## Interface
```solidity
pragma solidity >=0.6.12;
pragma experimental ABIEncoderV2;
interface ILogic {
    struct IExpert {
        uint32 createTime; //创建时间
        uint32 maturityTime; //到期时间
        uint256 subscriptionFee; //订阅费
        uint256 unclaimedIncome; //待领取订阅费
        uint256 earnedIncome; //已领取订阅费
    }

    //从0开始的专家总数量
    function expertTotal() external view returns (uint256);
    //续费周期（1=1年）对应的续费金额（1e18=1）
    function applyExpertFee(uint8 _cycle) external view returns (uint256);
    //专家状态
    function expert2Bool(address _expert) external view returns (bool);
    //用户订阅专家状态
    function expert2Subscribe(
        address _user,
        address _expert
    ) external view returns (bool);

    /**
    @dev 申请成为专家号，支付syt代币
    @param _cycle 申请周期 1=1年，注意判断资产是否足够
    @param _fee 用户订阅费用
     */
    function applyExpert(uint8 _cycle, uint256 _fee) external;

    // 管理员修改续费
    function updateApplyExpertFee(
        uint8 _cycle,
        uint256 _applyExpertFee
    ) external;

    // 修改自己的订阅费用
    function updateSubscriptionFee(uint256 _subscriptionFee) external;

    // 关注专家,支付syt代币
    function subscribe(address _expert) external;
    //   专家续费,支付syt代币
    function expertRenewal(uint8 _cycle) external;

    //    粉丝列表
    function getExpertFans(
        address _expert
    ) external view returns (address[] memory);
    /**
    @dev 订阅列表
     */
    function getFanSubscriptions(
        address _user
    ) external view returns (address[] memory);
    /**
    @dev 获取所有专家列表
     */
    function getAllExpert() external view returns (ExpertUser[] memory);
    /**
     @dev 领取自己的订阅费
      */
    function collectSubscriptionFee() external;

    // 获取单个专家信息
    function experts( address _expert) external view returns(ExpertUser memory);
}

```
